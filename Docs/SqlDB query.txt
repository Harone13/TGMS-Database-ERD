CREATE TABLE Sessions ( Id INT IDENTITY(1,1) PRIMARY KEY, Title NVARCHAR(200), CreatedAt DATETIME NOT NULL DEFAULT GETDATE(), PlannedStart DATETIME, PlannedEnd DATETIME, State INT NOT NULL, PlannedPomodoroCount INT DEFAULT 0, ActualPomodoroCount INT DEFAULT 0 );
CREATE TABLE SessionTasks ( Id INT IDENTITY(1,1) PRIMARY KEY, SessionId INT NOT NULL, TaskId INT NOT NULL, AllocatedPomodoros INT DEFAULT 0, AllocatedMinutes INT DEFAULT 0, Note NVARCHAR(MAX), FOREIGN KEY (SessionId) REFERENCES Sessions(Id) ON DELETE CASCADE, FOREIGN KEY (TaskId) REFERENCES Tasks(Id) ON DELETE NO ACTION);
CREATE INDEX idx_sessiontasks_session ON SessionTasks(SessionId); CREATE INDEX idx_sessiontasks_task ON SessionTasks(TaskId);
CREATE TABLE FocusPeriods ( Id INT IDENTITY(1,1) PRIMARY KEY, SessionId INT NOT NULL, PeriodIndex INT NOT NULL, Type INT NOT NULL, PlannedDurationMinutes INT NOT NULL, StartTime DATETIME, EndTime DATETIME, State INT NOT NULL, FOREIGN KEY (SessionId) REFERENCES Sessions(Id) ON DELETE CASCADE, CONSTRAINT CK_FocusPeriods_EndAfterStart CHECK (EndTime IS NULL OR EndTime > StartTime) ); CREATE INDEX idx_focusperiods_session ON FocusPeriods(SessionId);
CREATE TABLE PomodoroEvents ( Id INT IDENTITY(1,1) PRIMARY KEY, TaskId INT, SessionId INT, FocusPeriodId INT, StartTime DATETIME, EndTime DATETIME, MinutesSpent INT, State INT, FOREIGN KEY (TaskId) REFERENCES Tasks(Id), FOREIGN KEY (SessionId) REFERENCES Sessions(Id), FOREIGN KEY (FocusPeriodId) REFERENCES FocusPeriods(Id), CONSTRAINT CK_Events_EndAfterStart CHECK (EndTime IS NULL OR EndTime > StartTime) ); CREATE INDEX idx_pomodoroevents_task ON PomodoroEvents(TaskId); CREATE INDEX idx_pomodoroevents_session ON PomodoroEvents(SessionId);
